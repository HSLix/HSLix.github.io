# 代码解答（强烈建议自行解答后再看）
- 参考题解
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxLevelSum(self, root: Optional[TreeNode]) -> int:
        layer = [root]
        ans = 0
        cur_index = 1
        rec_s = -inf
        while layer:
            cur_s = 0
            nxt_layer = []
            for node in layer:
                cur_s += node.val
                if node.left:
                    nxt_layer.append(node.left)
                if node.right:
                    nxt_layer.append(node.right)
            if cur_s > rec_s:
                ans = cur_index
                rec_s = cur_s
                # print(cur_s)
            
            layer = nxt_layer
            cur_index += 1

        return ans
```
