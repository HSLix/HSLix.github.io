- 参考题解
```python
class Solution:
    def minDeletionSize(self, strs: List[str]) -> int:
        ans = 0
        for col in zip(*strs):
            if any(x > y for x, y in pairwise(col)):
                ans += 1
        return ans

        n = len(strs[0])
        is_up = [True]*n
        rec = ['a']*n
        for s in strs:
            for i in range(n):
                if is_up[i] and s[i] >= rec[i]:
                    rec[i] = s[i]
                else:
                    is_up[i] = False
        return is_up.count(False)
```
